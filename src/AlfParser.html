Expression ::= ( PotentiallyAmbiguousQualifiedName NameToExpressionCompletion | NonNameExpression )

NonIndexExpressionCompletion ::= ( Feature FeatureToExpressionCompletion | SequenceOperationOrReductionOrExpansion ExpressionCompletion | ConditionalExpressionCompletion )
NameToExpressionCompletion ::= ( PostfixOperationOrAssignment | NameToPrimaryExpressionCompletion ExpressionCompletion | Index ( PostfixOperationOrAssignment | ExpressionCompletion ) | NonIndexExpressionCompletion )
FeatureToExpressionCompletion ::= ( PostfixOperationOrAssignment | Index ( PostfixOperationOrAssignment | ExpressionCompletion ) | FeatureInvocation ExpressionCompletion | NonIndexExpressionCompletion )

ConditionalExpression ::= UnaryExpression ConditionalExpressionCompletion
ConditionalExpressionCompletion ::= ConditionalOrExpressionCompletion ( <HOOK> Expression <COLON> ConditionalExpression )? 
/* ASSIGNMENT EXPRESSIONS */
Assignment ::= AssignmentOperator Expression
LeftHandSide ::= ( PotentiallyAmbiguousQualifiedName NameToLeftHandSideCompletion | BaseExpression LeftHandSideCompletion | ParenthesizedExpression LeftHandSideCompletion )

NonIndexLeftHandSideCompletion ::= ( Feature FeatureToLeftHandSideCompletion | SequenceOperationOrReductionOrExpansion LeftHandSideCompletion )
NameToLeftHandSideCompletion ::= ( NameToPrimaryExpressionCompletion LeftHandSideCompletion | Index ( LeftHandSideCompletion )? | NonIndexLeftHandSideCompletion )?
FeatureToLeftHandSideCompletion ::= ( Index ( LeftHandSideCompletion )? | FeatureInvocation LeftHandSideCompletion | NonIndexLeftHandSideCompletion )?
AssignmentOperator ::= ( <ASSIGN> | <PLUSASSIGN> | <MINUSASSIGN> | <STARASSIGN> | <SLASHASSIGN> | <REMASSIGN> | <ANDASSIGN> | <ORASSIGN> | <XORASSIGN> )

/* IF STATEMENTS */

/* SWITCH STATEMENTS */
SwitchStatement ::= <SWITCH> <LPAREN> Expression <RPAREN> <LBRACE> ( SwitchClause )* ( SwitchDefaultClause )? <RBRACE>
